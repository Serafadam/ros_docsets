326  template<typename SubContext, typename ... Args>
327  std::shared_ptr<SubContext>
328  get_sub_context(Args && ... args)
329  {
330  std::lock_guard<std::recursive_mutex> lock(sub_contexts_mutex_);
331 
332  std::type_index type_i(typeid(SubContext));
333  std::shared_ptr<SubContext> sub_context;
334  auto it = sub_contexts_.find(type_i);
335  if (it == sub_contexts_.end()) {
336  // It doesn't exist yet, make it
337  sub_context = std::shared_ptr<SubContext>(
338  new SubContext(std::forward<Args>(args) ...),
339  [](SubContext * sub_context_ptr) {
340  delete sub_context_ptr;
341  });
342  sub_contexts_[type_i] = sub_context;
343  } else {
344  // It exists, get it out and cast it.
345  sub_context = std::static_pointer_cast<SubContext>(it->second);
346  }
347  return sub_context;
348  }
