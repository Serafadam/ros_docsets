421  std::shared_future<typename GoalHandle::SharedPtr>
422  async_send_goal(const Goal & goal, const SendGoalOptions & options = SendGoalOptions())
423  {
424  // Put promise in the heap to move it around.
425  auto promise = std::make_shared<std::promise<typename GoalHandle::SharedPtr>>();
426  std::shared_future<typename GoalHandle::SharedPtr> future(promise->get_future());
427  using GoalRequest = typename ActionT::Impl::SendGoalService::Request;
428  auto goal_request = std::make_shared<GoalRequest>();
429  goal_request->goal_id.uuid = this->generate_goal_id();
430  goal_request->goal = goal;
431  this->send_goal_request(
432  std::static_pointer_cast<void>(goal_request),
433  [this, goal_request, options, promise](std::shared_ptr<void> response) mutable
434  {
435  using GoalResponse = typename ActionT::Impl::SendGoalService::Response;
436  auto goal_response = std::static_pointer_cast<GoalResponse>(response);
437  if (!goal_response->accepted) {
438  promise->set_value(nullptr);
439  if (options.goal_response_callback) {
440  options.goal_response_callback(nullptr);
441  }
442  return;
443  }
444  GoalInfo goal_info;
445  goal_info.goal_id.uuid = goal_request->goal_id.uuid;
446  goal_info.stamp = goal_response->stamp;
447  // Do not use std::make_shared as friendship cannot be forwarded.
448  std::shared_ptr<GoalHandle> goal_handle(
449  new GoalHandle(goal_info, options.feedback_callback, options.result_callback));
450  {
451  std::lock_guard<std::mutex> guard(goal_handles_mutex_);
452  goal_handles_[goal_handle->get_goal_id()] = goal_handle;
453  }
454  promise->set_value(goal_handle);
455  if (options.goal_response_callback) {
456  options.goal_response_callback(goal_handle);
457  }
458 
459  if (options.result_callback) {
460  this->make_result_aware(goal_handle);
461  }
462  });
463 
464  // TODO(jacobperron): Encapsulate into it's own function and
465  // consider exposing an option to disable this cleanup
466  // To prevent the list from growing out of control, forget about any goals
467  // with no more user references
468  {
469  std::lock_guard<std::mutex> guard(goal_handles_mutex_);
470  auto goal_handle_it = goal_handles_.begin();
471  while (goal_handle_it != goal_handles_.end()) {
472  if (!goal_handle_it->second.lock()) {
473  RCLCPP_DEBUG(
474  this->get_logger(),
475  "Dropping weak reference to goal handle during send_goal()");
476  goal_handle_it = goal_handles_.erase(goal_handle_it);
477  } else {
478  ++goal_handle_it;
479  }
480  }
481  }
482 
483  return future;
484  }

494  std::shared_future<WrappedResult>
495  async_get_result(
496  typename GoalHandle::SharedPtr goal_handle,
497  ResultCallback result_callback = nullptr)
498  {
499  std::lock_guard<std::mutex> lock(goal_handles_mutex_);
500  if (goal_handles_.count(goal_handle->get_goal_id()) == 0) {
501  throw exceptions::UnknownGoalHandleError();
502  }
503  if (goal_handle->is_invalidated()) {
504  // This case can happen if there was a failure to send the result request
505  // during the goal response callback
506  throw goal_handle->invalidate_exception_;
507  }
508  if (result_callback) {
509  // This will override any previously registered callback
510  goal_handle->set_result_callback(result_callback);
511  }
512  this->make_result_aware(goal_handle);
513  return goal_handle->async_get_result();
514  }

637  void
638  handle_feedback_message(std::shared_ptr<void> message) override
639  {
640  std::lock_guard<std::mutex> guard(goal_handles_mutex_);
641  using FeedbackMessage = typename ActionT::Impl::FeedbackMessage;
642  typename FeedbackMessage::SharedPtr feedback_message =
643  std::static_pointer_cast<FeedbackMessage>(message);
644  const GoalUUID & goal_id = feedback_message->goal_id.uuid;
645  if (goal_handles_.count(goal_id) == 0) {
646  RCLCPP_DEBUG(
647  this->get_logger(),
648  "Received feedback for unknown goal. Ignoring...");
649  return;
650  }
651  typename GoalHandle::SharedPtr goal_handle = goal_handles_[goal_id].lock();
652  // Forget about the goal if there are no more user references
653  if (!goal_handle) {
654  RCLCPP_DEBUG(
655  this->get_logger(),
656  "Dropping weak reference to goal handle during feedback callback");
657  goal_handles_.erase(goal_id);
658  return;
659  }
660  auto feedback = std::make_shared<Feedback>();
661  *feedback = feedback_message->feedback;
662  goal_handle->call_feedback_callback(goal_handle, feedback);
663  }

674  void
675  handle_status_message(std::shared_ptr<void> message) override
676  {
677  std::lock_guard<std::mutex> guard(goal_handles_mutex_);
678  using GoalStatusMessage = typename ActionT::Impl::GoalStatusMessage;
679  auto status_message = std::static_pointer_cast<GoalStatusMessage>(message);
680  for (const GoalStatus & status : status_message->status_list) {
681  const GoalUUID & goal_id = status.goal_info.goal_id.uuid;
682  if (goal_handles_.count(goal_id) == 0) {
683  RCLCPP_DEBUG(
684  this->get_logger(),
685  "Received status for unknown goal. Ignoring...");
686  continue;
687  }
688  typename GoalHandle::SharedPtr goal_handle = goal_handles_[goal_id].lock();
689  // Forget about the goal if there are no more user references
690  if (!goal_handle) {
691  RCLCPP_DEBUG(
692  this->get_logger(),
693  "Dropping weak reference to goal handle during status callback");
694  goal_handles_.erase(goal_id);
695  continue;
696  }
697  goal_handle->set_status(status.status);
698  }

702  void
703  make_result_aware(typename GoalHandle::SharedPtr goal_handle)
704  {
705  // Avoid making more than one request
706  if (goal_handle->set_result_awareness(true)) {
707  return;
708  }
709  using GoalResultRequest = typename ActionT::Impl::GetResultService::Request;
710  auto goal_result_request = std::make_shared<GoalResultRequest>();
711  goal_result_request->goal_id.uuid = goal_handle->get_goal_id();
712  try {
713  this->send_result_request(
714  std::static_pointer_cast<void>(goal_result_request),
715  [goal_handle, this](std::shared_ptr<void> response) mutable
716  {
717  // Wrap the response in a struct with the fields a user cares about
718  WrappedResult wrapped_result;
719  using GoalResultResponse = typename ActionT::Impl::GetResultService::Response;
720  auto result_response = std::static_pointer_cast<GoalResultResponse>(response);
721  wrapped_result.result = std::make_shared<typename ActionT::Result>();
722  *wrapped_result.result = result_response->result;
723  wrapped_result.goal_id = goal_handle->get_goal_id();
724  wrapped_result.code = static_cast<ResultCode>(result_response->status);
725  goal_handle->set_result(wrapped_result);
726  std::lock_guard<std::mutex> lock(goal_handles_mutex_);
727  goal_handles_.erase(goal_handle->get_goal_id());
728  });
729  } catch (rclcpp::exceptions::RCLError & ex) {
730  // This will cause an exception when the user tries to access the result
731  goal_handle->invalidate(exceptions::UnawareGoalHandleError(ex.message));
732  }
733  }
