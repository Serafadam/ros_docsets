410  std::pair<GoalResponse, std::shared_ptr<void>>
411  call_handle_goal_callback(GoalUUID & uuid, std::shared_ptr<void> message) override
412  {
413  auto request = std::static_pointer_cast<
414  typename ActionT::Impl::SendGoalService::Request>(message);
415  auto goal = std::shared_ptr<typename ActionT::Goal>(request, &request->goal);
416  GoalResponse user_response = handle_goal_(uuid, goal);
417 
418  auto ros_response = std::make_shared<typename ActionT::Impl::SendGoalService::Response>();
419  ros_response->accepted = GoalResponse::ACCEPT_AND_EXECUTE == user_response ||
420  GoalResponse::ACCEPT_AND_DEFER == user_response;
421  return std::make_pair(user_response, ros_response);
422  }
423 
425  CancelResponse
426  call_handle_cancel_callback(const GoalUUID & uuid) override
427  {
428  std::shared_ptr<ServerGoalHandle<ActionT>> goal_handle;
429  {
430  std::lock_guard<std::mutex> lock(goal_handles_mutex_);
431  auto element = goal_handles_.find(uuid);
432  if (element != goal_handles_.end()) {
433  goal_handle = element->second.lock();
434  }
435  }
436 
437  CancelResponse resp = CancelResponse::REJECT;
438  if (goal_handle) {
439  resp = handle_cancel_(goal_handle);
440  if (CancelResponse::ACCEPT == resp) {
441  try {
442  goal_handle->_cancel_goal();
443  } catch (const rclcpp::exceptions::RCLError & ex) {
444  RCLCPP_DEBUG(
445  rclcpp::get_logger("rclcpp_action"),
446  "Failed to cancel goal in call_handle_cancel_callback: %s", ex.what());
447  return CancelResponse::REJECT;
448  }
449  }
450  }
451  return resp;
452  }

455  void
456  call_goal_accepted_callback(
457  std::shared_ptr<rcl_action_goal_handle_t> rcl_goal_handle,
458  GoalUUID uuid, std::shared_ptr<void> goal_request_message) override
459  {
460  std::shared_ptr<ServerGoalHandle<ActionT>> goal_handle;
461  std::weak_ptr<Server<ActionT>> weak_this = this->shared_from_this();
462 
463  std::function<void(const GoalUUID &, std::shared_ptr<void>)> on_terminal_state =
464  [weak_this](const GoalUUID & goal_uuid, std::shared_ptr<void> result_message)
465  {
466  std::shared_ptr<Server<ActionT>> shared_this = weak_this.lock();
467  if (!shared_this) {
468  return;
469  }
470  // Send result message to anyone that asked
471  shared_this->publish_result(goal_uuid, result_message);
472  // Publish a status message any time a goal handle changes state
473  shared_this->publish_status();
474  // notify base so it can recalculate the expired goal timer
475  shared_this->notify_goal_terminal_state();
476  // Delete data now (ServerBase and rcl_action_server_t keep data until goal handle expires)
477  std::lock_guard<std::mutex> lock(shared_this->goal_handles_mutex_);
478  shared_this->goal_handles_.erase(goal_uuid);
479  };
480 
481  std::function<void(const GoalUUID &)> on_executing =
482  [weak_this](const GoalUUID & goal_uuid)
483  {
484  std::shared_ptr<Server<ActionT>> shared_this = weak_this.lock();
485  if (!shared_this) {
486  return;
487  }
488  (void)goal_uuid;
489  // Publish a status message any time a goal handle changes state
490  shared_this->publish_status();
491  };
492 
493  std::function<void(std::shared_ptr<typename ActionT::Impl::FeedbackMessage>)> publish_feedback =
494  [weak_this](std::shared_ptr<typename ActionT::Impl::FeedbackMessage> feedback_msg)
495  {
496  std::shared_ptr<Server<ActionT>> shared_this = weak_this.lock();
497  if (!shared_this) {
498  return;
499  }
500  shared_this->publish_feedback(std::static_pointer_cast<void>(feedback_msg));
501  };
502 
503  auto request = std::static_pointer_cast<
504  const typename ActionT::Impl::SendGoalService::Request>(goal_request_message);
505  auto goal = std::shared_ptr<const typename ActionT::Goal>(request, &request->goal);
506  goal_handle.reset(
507  new ServerGoalHandle<ActionT>(
508  rcl_goal_handle, uuid, goal, on_terminal_state, on_executing, publish_feedback));
509  {
510  std::lock_guard<std::mutex> lock(goal_handles_mutex_);
511  goal_handles_[uuid] = goal_handle;
512  }
513  handle_accepted_(goal_handle);
514  }
