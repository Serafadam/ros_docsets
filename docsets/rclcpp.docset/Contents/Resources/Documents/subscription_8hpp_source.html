130  Subscription(
131  rclcpp::node_interfaces::NodeBaseInterface * node_base,
132  const rosidl_message_type_support_t & type_support_handle,
133  const std::string & topic_name,
134  const rclcpp::QoS & qos,
135  AnySubscriptionCallback<MessageT, AllocatorT> callback,
136  const rclcpp::SubscriptionOptionsWithAllocator<AllocatorT> & options,
137  typename MessageMemoryStrategyT::SharedPtr message_memory_strategy,
138  SubscriptionTopicStatisticsSharedPtr subscription_topic_statistics = nullptr)
139  : SubscriptionBase(
140  node_base,
141  type_support_handle,
142  topic_name,
143  options.template to_rcl_subscription_options<ROSMessageType>(qos),
144  callback.is_serialized_message_callback()),
145  any_callback_(callback),
146  options_(options),
147  message_memory_strategy_(message_memory_strategy)
148  {
149  if (options_.event_callbacks.deadline_callback) {
150  this->add_event_handler(
151  options_.event_callbacks.deadline_callback,
152  RCL_SUBSCRIPTION_REQUESTED_DEADLINE_MISSED);
153  }
154  if (options_.event_callbacks.liveliness_callback) {
155  this->add_event_handler(
156  options_.event_callbacks.liveliness_callback,
157  RCL_SUBSCRIPTION_LIVELINESS_CHANGED);
158  }
159  if (options_.event_callbacks.incompatible_qos_callback) {
160  this->add_event_handler(
161  options_.event_callbacks.incompatible_qos_callback,
162  RCL_SUBSCRIPTION_REQUESTED_INCOMPATIBLE_QOS);
163  } else if (options_.use_default_callbacks) {
164  // Register default callback when not specified
165  try {
166  this->add_event_handler(
167  [this](QOSRequestedIncompatibleQoSInfo & info) {
168  this->default_incompatible_qos_callback(info);
169  },
170  RCL_SUBSCRIPTION_REQUESTED_INCOMPATIBLE_QOS);
171  } catch (UnsupportedEventTypeException & /*exc*/) {
172  // pass
173  }
174  }
175  if (options_.event_callbacks.message_lost_callback) {
176  this->add_event_handler(
177  options_.event_callbacks.message_lost_callback,
178  RCL_SUBSCRIPTION_MESSAGE_LOST);
179  }
180 
181  // Setup intra process publishing if requested.
182  if (rclcpp::detail::resolve_use_intra_process(options_, *node_base)) {
183  using rclcpp::detail::resolve_intra_process_buffer_type;
184 
185  // Check if the QoS is compatible with intra-process.
186  auto qos_profile = get_actual_qos();
187  if (qos_profile.history() != rclcpp::HistoryPolicy::KeepLast) {
188  throw std::invalid_argument(
189  "intraprocess communication allowed only with keep last history qos policy");
190  }
191  if (qos_profile.depth() == 0) {
192  throw std::invalid_argument(
193  "intraprocess communication is not allowed with 0 depth qos policy");
194  }
195  if (qos_profile.durability() != rclcpp::DurabilityPolicy::Volatile) {
196  throw std::invalid_argument(
197  "intraprocess communication allowed only with volatile durability");
198  }
199 
200  using SubscriptionIntraProcessT = rclcpp::experimental::SubscriptionIntraProcess<
201  MessageT,
202  SubscribedType,
203  SubscribedTypeAllocator,
204  SubscribedTypeDeleter,
205  ROSMessageT,
206  AllocatorT>;
207 
208  // First create a SubscriptionIntraProcess which will be given to the intra-process manager.
209  auto context = node_base->get_context();
210  subscription_intra_process_ = std::make_shared<SubscriptionIntraProcessT>(
211  callback,
212  options_.get_allocator(),
213  context,
214  this->get_topic_name(), // important to get like this, as it has the fully-qualified name
215  qos_profile,
216  resolve_intra_process_buffer_type(options_.intra_process_buffer_type, callback));
217  TRACEPOINT(
218  rclcpp_subscription_init,
219  static_cast<const void *>(get_subscription_handle().get()),
220  static_cast<const void *>(subscription_intra_process_.get()));
221 
222  // Add it to the intra process manager.
223  using rclcpp::experimental::IntraProcessManager;
224  auto ipm = context->get_sub_context<IntraProcessManager>();
225  uint64_t intra_process_subscription_id = ipm->add_subscription(subscription_intra_process_);
226  this->setup_intra_process(intra_process_subscription_id, ipm);
227  }
228 
229  if (subscription_topic_statistics != nullptr) {
230  this->subscription_topic_statistics_ = std::move(subscription_topic_statistics);
231  }
232 
233  TRACEPOINT(
234  rclcpp_subscription_init,
235  static_cast<const void *>(get_subscription_handle().get()),
236  static_cast<const void *>(this));
237  TRACEPOINT(
238  rclcpp_subscription_callback_added,
239  static_cast<const void *>(this),
240  static_cast<const void *>(&any_callback_));
241  // The callback object gets copied, so if registration is done too early/before this point
242  // (e.g. in `AnySubscriptionCallback::set()`), its address won't match any address used later
243  // in subsequent tracepoints.
244 #ifndef TRACETOOLS_DISABLED
245  any_callback_.register_callback_for_tracing();
246 #endif

324  void
325  handle_message(
326  std::shared_ptr<void> & message,
327  const rclcpp::MessageInfo & message_info) override
328  {
329  if (matches_any_intra_process_publishers(&message_info.get_rmw_message_info().publisher_gid)) {
330  // In this case, the message will be delivered via intra process and
331  // we should ignore this copy of the message.
332  return;
333  }
334  auto typed_message = std::static_pointer_cast<ROSMessageType>(message);
335 
336  std::chrono::time_point<std::chrono::system_clock> now;
337  if (subscription_topic_statistics_) {
338  // get current time before executing callback to
339  // exclude callback duration from topic statistics result.
340  now = std::chrono::system_clock::now();
341  }
342 
343  any_callback_.dispatch(typed_message, message_info);
344 
345  if (subscription_topic_statistics_) {
346  const auto nanos = std::chrono::time_point_cast<std::chrono::nanoseconds>(now);
347  const auto time = rclcpp::Time(nanos.time_since_epoch().count());
348  subscription_topic_statistics_->handle_message(*typed_message, time);
349  }
350  }

361  void
362  handle_loaned_message(
363  void * loaned_message,
364  const rclcpp::MessageInfo & message_info) override
365  {
366  if (matches_any_intra_process_publishers(&message_info.get_rmw_message_info().publisher_gid)) {
367  // In this case, the message will be delivered via intra process and
368  // we should ignore this copy of the message.
369  return;
370  }
371 
372  auto typed_message = static_cast<ROSMessageType *>(loaned_message);
373  // message is loaned, so we have to make sure that the deleter does not deallocate the message
374  auto sptr = std::shared_ptr<ROSMessageType>(
375  typed_message, [](ROSMessageType * msg) {(void) msg;});
376 
377  std::chrono::time_point<std::chrono::system_clock> now;
378  if (subscription_topic_statistics_) {
379  // get current time before executing callback to
380  // exclude callback duration from topic statistics result.
381  now = std::chrono::system_clock::now();
382  }
383 
384  any_callback_.dispatch(sptr, message_info);
385 
386  if (subscription_topic_statistics_) {
387  const auto nanos = std::chrono::time_point_cast<std::chrono::nanoseconds>(now);
388  const auto time = rclcpp::Time(nanos.time_since_epoch().count());
389  subscription_topic_statistics_->handle_message(*typed_message, time);
390  }
